import socket

def get_internal_ip():
    try:
        # 创建一个socket连接
        s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        # 连接到一个外部地址（这里使用Google的DNS服务器）
        s.connect(("8.8.8.8", 80))
        # 获取本地socket的IP地址
        internal_ip = s.getsockname()[0]
        s.close()
        return internal_ip
    except Exception as e:
        return f"无法获取内网IP: {str(e)}"

if __name__ == "__main__":
    ip = get_internal_ip()
    print(f"内网IP地址: {ip}")
